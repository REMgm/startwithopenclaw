The Universal Connector Pattern

Every API integration follows the same structure. Master this pattern once, apply it everywhere:

# The 4-Step Integration Pattern
1. Authenticate   → Get API key/OAuth token
2. Request      → Send structured API call
3. Parse        → Extract relevant data
4. Act          → Do something with results

# OpenClaw makes this trivial:
@agent.tool
def get_gmail_inbox(credentials, query=""):
    # Step 1: Auth (handled by CLAW)
    service = build('gmail', 'v1', credentials=credentials)
    
    # Step 2: Request
    results = service.users().messages().list(
        userId='me', q=query, maxResults=10
    ).execute()
    
    # Step 3: Parse
    messages = results.get('messages', [])
    
    # Step 4: Return for agent to use
    return format_messages(messages)

Gmail Integration Deep Dive

Let's build a complete Gmail integration. Your agent will read emails, send responses, and manage labels.

# Save as skills/gmail_integration.py
from google.oauth2.credentials import Credentials
from googleapiclient.discovery import build
from email.mime.text import MIMEText
import base64

class GmailClient:
    def __init__(self, token_json):
        creds = Credentials.from_authorized_user_info(token_json)
        self.service = build('gmail', 'v1', credentials=creds)
    
    def get_unread(self, max_results=10):
        """Fetch unread emails"""
        results = self.service.users().messages().list(
            userId='me', 
            q='is:unread',
            maxResults=max_results
        ).execute()
        return results.get('messages', [])
    
    def read_message(self, msg_id):
        """Get full message content"""
        message = self.service.users().messages().get(
            userId='me', id=msg_id, format='full'
        ).execute()
        
        headers = message['payload']['headers']
        subject = next(h['value'] for h in headers if h['name'] == 'Subject')
        sender = next(h['value'] for h in headers if h['name'] == 'From')
        
        # Extract body
        if 'parts' in message['payload']:
            body = message['payload']['parts'][0]['body'].get('data', '')
        else:
            body = message['payload']['body'].get('data', '')
        
        return {
            'id': msg_id,
            'subject': subject,
            'from': sender,
            'body': base64.urlsafe_b64decode(body).decode('utf-8') if body else ''
        }
    
    def send_reply(self, original_msg_id, reply_text):
        """Send a reply to an email"""
        # Get original thread
        original = self.service.users().messages().get(
            userId='me', id=original_msg_id
        ).execute()
        
        thread_id = original['threadId']
        headers = original['payload']['headers']
        to = next(h['value'] for h in headers if h['name'] == 'From')
        subject = next(h['value'] for h in headers if h['name'] == 'Subject')
        
        # Create reply
        message = MIMEText(reply_text)
        message['to'] = to
        message['subject'] = f"Re: {subject.replace('Re: ', '')}"
        message['In-Reply-To'] = original_msg_id
        
        raw = base64.urlsafe_b64encode(message.as_bytes()).decode()
        
        self.service.users().messages().send(
            userId='me',
            body={'raw': raw, 'threadId': thread_id}
        ).execute()

Natural Language Control

Now wire it into your agent with natural language commands:

@agent.on_message
def handle_email_commands(ctx, message):
    gmail = GmailClient(ctx.user.tokens['gmail'])
    
    if "check my email" in message.lower():
        unread = gmail.get_unread(max_results=5)
        
        if not unread:
            return "📧 No unread emails!"
        
        response = f"📧 {len(unread)} unread emails:\n\n"
        
        for msg in unread[:3]:  # Summarize top 3
            email = gmail.read_message(msg['id'])
            response += f"• **{email['subject']}** from {email['from']}\n"
            response += f"  _{email['body'][:100]}..._\n\n"
        
        return response
    
    elif message.lower().startswith("reply to"):
        # Parse: "reply to [subject] saying [message]"
        parts = message.replace("reply to ", "").split(" saying ", 1)
        if len(parts) == 2:
            subject_query, reply_text = parts
            
            # Find the email
            results = gmail.service.users().messages().list(
                userId='me', q=f"subject:{subject_query}", maxResults=1
            ).execute()
            
            if results['messages']:
                msg_id = results['messages'][0]['id']
                gmail.send_reply(msg_id, reply_text)
                return f"✅ Reply sent to {subject_query}"
            else:
                return f"❌ Couldn't find email with subject: {subject_query}"

Slack Integration

Connect your agent to team communication:

import slack_sdk

class SlackClient:
    def __init__(self, bot_token):
        self.client = slack_sdk.WebClient(token=bot_token)
    
    def post_message(self, channel, text, blocks=None):
        """Post to any channel"""
        return self.client.chat_postMessage(
            channel=channel,
            text=text,
            blocks=blocks
        )
    
    def read_messages(self, channel, limit=10):
        """Read recent messages"""
        result = self.client.conversations_history(
            channel=channel,
            limit=limit
        )
        return result['messages']
    
    def upload_file(self, channel, file_path, title=None):
        """Share files"""
        return self.client.files_upload_v2(
            channel=channel,
            file=file_path,
            title=title
        )

Notion Integration

from notion_client import Client

class NotionClient:
    def __init__(self, token):
        self.client = Client(auth=token)
    
    def create_page(self, parent_id, title, content):
        """Create a new Notion page"""
        return self.client.pages.create(
            parent={"database_id": parent_id},
            properties={
                "Name": {"title": [{"text": {"content": title}}]}
            },
            children=[{
                "object": "block",
                "type": "paragraph",
                "paragraph": {
                    "rich_text": [{"text": {"content": content}}]
                }
            }]
        )
    
    def query_database(self, database_id, filter_obj=None):
        """Search and filter databases"""
        return self.client.databases.query(
            database_id=database_id,
            filter=filter_obj
        )

GitHub Integration

import requests

class GitHubClient:
    def __init__(self, token):
        self.headers = {
            "Authorization": f"token {token}",
            "Accept": "application/vnd.github.v3+json"
        }
    
    def get_issues(self, repo, state="open"):
        """List issues for a repository"""
        url = f"https://api.github.com/repos/{repo}/issues"
        params = {"state": state}
        response = requests.get(url, headers=self.headers, params=params)
        return response.json()
    
    def create_issue(self, repo, title, body, labels=None):
        """Create a new issue"""
        url = f"https://api.github.com/repos/{repo}/issues"
        data = {"title": title, "body": body, "labels": labels or []}
        response = requests.post(url, headers=self.headers, json=data)
        return response.json()
    
    def get_pull_requests(self, repo, state="open"):
        """List PRs"""
        url = f"https://api.github.com/repos/{repo}/pulls"
        params = {"state": state}
        response = requests.get(url, headers=self.headers, params=params)
        return response.json()

The Webhook Superpower

APIs let your agent pull data. Webhooks let services push data to your agent. This enables real-time reactions.

# Set up a webhook endpoint in OpenClaw
from flask import Flask, request

app = Flask(__name__)

@app.route('/webhook/stripe', methods=['POST'])
def handle_stripe_payment():
    """Receive real-time payment events"""
    event = request.json
    
    if event['type'] == 'payment_intent.succeeded':
        payment = event['data']['object']
        
        # Notify user
        agent.send_message(f"""
💰 New payment received!
Amount: ${payment['amount'] / 100}
Customer: {payment['customer']}
Status: ✅ Succeeded
        """)
        
        # Add to Notion CRM
        notion.create_page(
            database_id="payments_db",
            title=f"Payment from {payment['customer']}",
            content=str(payment)
        )
    
    return "OK", 200
